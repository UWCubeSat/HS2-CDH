26bool loadBitstreamIntoBuffer(const char* bitstream_path,
27 std::uint8_t* bitstream_buf,
28 const std::size_t buffer_capacity,
29 std::size_t& bytes_loaded) {
30 bytes_loaded = 0U;
31 FILE* file = std::fopen(bitstream_path, "rb");
32 if (file == nullptr) {
33 return false;
34 }
35
36 const std::size_t bytes_read = std::fread(bitstream_buf, 1U, buffer_capacity, file);
37 const bool file_truncated = (bytes_read == buffer_capacity) && (std::fgetc(file) != EOF);
38 std::fclose(file);
39
40 if (file_truncated || (bytes_read == 0U)) {
41 return false;
42 }
43
44 bytes_loaded = bytes_read;
45 return true;
46}

48std::string buildOutputFilePath(const char* input_path, const char* output_dir) {
49 if ((input_path == nullptr) || (output_dir == nullptr)) { // GCOVR_EXCL_LINE
50 return std::string(); // GCOVR_EXCL_LINE
51 }
52
53 std::string dir(output_dir);
54 if (!dir.empty() && dir.back() != '/') { // GCOVR_EXCL_BR_LINE
55 dir.push_back('/');
56 }
57
58 std::string file(input_path);
59 const std::size_t slash = file.find_last_of('/');
60 if (slash != std::string::npos) {
61 file = file.substr(slash + 1U);

63

66 file = file.substr(0U, dot);
