26std::string buildOutputFilePath(const char* input_path, const char* output_dir) {
27 if ((input_path == nullptr) || (output_dir == nullptr)) { // GCOVR_EXCL_LINE
28 return std::string(); // GCOVR_EXCL_LINE
29 }
30
31 std::string dir(output_dir);
32 if (!dir.empty() && dir.back() != '/') { // GCOVR_EXCL_BR_LINE
33 dir.push_back('/');
34 }
35
36 std::string file(input_path);
37 const std::size_t slash = file.find_last_of('/');
38 if (slash != std::string::npos) {
39 file = file.substr(slash + 1U);
40 }
41
42 const std::size_t dot = file.find_last_of('.');
43 if (dot != std::string::npos) {
44 file = file.substr(0U, dot);
45 }
46 return dir + file;
47}
