25std::string buildOutputFilePath(const char* input_path, const char* output_dir) {
26 if ((input_path == nullptr) || (output_dir == nullptr)) {
27 return std::string();
28 }
29
30 std::string dir(output_dir);
31 if (!dir.empty() && dir.back() != '/') {
32 dir.push_back('/');
33 }
34
35 std::string file(input_path);
36 const std::size_t slash = file.find_last_of('/');
37 if (slash != std::string::npos) {
38 file = file.substr(slash + 1U);
39 }
40
41 const std::size_t dot = file.find_last_of('.');
42 if (dot != std::string::npos) {
43 file = file.substr(0U, dot);
44 }
45
46 file += ".bin";
47 return dir + file;
48}
